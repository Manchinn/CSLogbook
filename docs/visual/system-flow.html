{
  "jti": "crypto.randomUUID()",    // Token ID for blacklist
  "userId": 1,                      // Always present
  "role": "teacher",                // student | teacher | admin
  // Student only:
  "studentCode": "6501234", "studentId": 10,
  // Teacher only:
  "teacherId": 5, "teacherCode": "T001", "teacherType": "support",
  "teacherPosition": "อาจารย์", "canAccessTopicExam": true, "canExportProject1": false,
  // Admin only:
  "adminId": 1,
  "iat": 1712700000, "exp": 1712786400
}
// ไม่มี PII: firstName, lastName, email — ดึงจาก verify-token endpoint แทน
